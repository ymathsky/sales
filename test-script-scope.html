// WITHOUT IIFE (Global Scope - ERROR)
let cameraStream = null;  // First execution: OK
let cameraStream = null;  // Second execution: ❌ ERROR!
                          // "cameraStream already declared"

// WITH IIFE (Function Scope - OK)
(function() {
    let cameraStream = null;  // First execution: OK
})();
(function() {
    let cameraStream = null;  // Second execution: ✓ OK!
})();                         // Each has isolated scope
